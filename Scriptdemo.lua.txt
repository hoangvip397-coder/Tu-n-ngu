-- ==================== XỬ LÝ SỰ KIỆN NÚT BẤM ====================

-- Khi người dùng bấm nút Get Key
GetKeyBtn.MouseButton1Click:Connect(function()
    if setclipboard then
        setclipboard(LINK_GET_KEY)
        StatusText.Text = "✅ Đã sao chép link Get Key vào bộ nhớ tạm!"
        StatusText.TextColor3 = Color3.fromRGB(46, 204, 113)
    else
        StatusText.Text = "❌ Thiết bị không hỗ trợ tự sao chép link."
        StatusText.TextColor3 = Color3.fromRGB(231, 76, 60)
    end
end)

-- Khi người dùng bấm nút Check Key
CheckKeyBtn.MouseButton1Click:Connect(function()
    local enteredKey = KeyBox.Text
    
    if enteredKey == KEY_CHIEU_KHOA then
        StatusText.Text = "✅ Đang kích hoạt Script chính..."
        StatusText.TextColor3 = Color3.fromRGB(46, 204, 113)
        task.wait(1)
        
        -- Xóa bảng nhập Key đi
        KeySystemGui:Destroy()
        
        -- ========================================================
        -- [SCRIPT CHÍNH CỦA BẠN SẼ CHẠY TỪ ĐÂY]
        print("Kích hoạt thành công!")
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Kích Hoạt Thành Công!",
            Text = "Chào mừng bạn đã quay trở lại.",
            Duration = 5
        })
        -- ========================================================
    else
        StatusText.Text = "❌ Key sai rồi! Vui lòng thử lại."
        StatusText.TextColor3 = Color3.fromRGB(231, 76, 60)
        KeyBox.Text = "" -- Xóa chữ nhập sai
    end
end)
